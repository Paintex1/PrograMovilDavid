const val PI: Double = 3.14159265359
fun main() {
    val radio: Double = 5.0  // Radio del círculo

    // Calcular el perímetro del círculo
    val perimetro: Double = 2 * PI * radio

    println("El perímetro del círculo es: $perimetro")
    
}
